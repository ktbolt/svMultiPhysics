50/// @brief Base class for boundary conditions with spatially variable arrays
51///
52/// This class provides common functionality for boundary conditions that need to
53/// read and manage arrays of values from VTP files or uniform values. It handles
54/// distribution of data across processes and provides efficient access to values.
55///
56/// This class is intended to be subclassed by specific boundary condition types.
57///
58/// Development note: this class is intended to eventually be an object-oriented
59/// replacement of the existing bcType, although it is not yet complete.
60///
61/// Example usage:
62/// ```cpp
63/// class MyBoundaryCondition : public BoundaryCondition {
64/// public:
65/// MyBoundaryCondition(const std::string& vtp_file_path, const std::vector<std::string>& array_names, const faceType& face)
66/// : BoundaryCondition(vtp_file_path, array_names, face) {}
67///
68/// MyBoundaryCondition(const std::map<std::string, double>& uniform_values, const faceType& face)
69/// : BoundaryCondition(uniform_values, face) {}
70///
71/// // Add BC-specific functionality here
72/// };
73/// ```
74class BoundaryCondition {
75protected:
76 /// @brief Type alias for map of array names to array data
77 using StringArrayMap = std::map<std::string, Array<double>>;
78 using StringBoolMap = std::map<std::string, bool>;
79 using StringDoubleMap = std::map<std::string, double>;
80
81 /// @brief Data members for BC
82 const faceType* face_ = nullptr; ///< Face associated with the BC (can be null)
83 int global_num_nodes_ = 0; ///< Global number of nodes on the face
84 int local_num_nodes_ = 0; ///< Local number of nodes on this processor
85 std::vector<std::string> array_names_; ///< Names of arrays to read from VTP file
86 StringArrayMap local_data_; ///< Local array values for each node on this processor
87 StringArrayMap global_data_; ///< Global array values (only populated on master)
88 StringBoolMap flags_; ///< Named boolean flags for BC behavior
89 bool spatially_variable = false; ///< Flag indicating if data is from VTP file
90 std::string vtp_file_path_; ///< Path to VTP file (empty if uniform)
91 std::map<int, int> global_node_map_; ///< Maps global node IDs to local array indices
92 std::unique_ptr<VtkVtpData> vtp_data_; ///< VTP data object
93
94public:
95 /// @brief Tolerance for point matching in VTP files
96 static constexpr double POINT_MATCH_TOLERANCE = 1e-12;
97
98 /// @brief Default constructor - creates an empty BC
99 BoundaryCondition() = default;
100
101 /// @brief Constructor - reads data from VTP file
102 /// @param vtp_file_path Path to VTP file containing arrays
103 /// @param array_names Names of arrays to read from VTP file
104 /// @param face Face associated with the BC
105 /// @throws std::runtime_error if file cannot be read or arrays are missing
106 BoundaryCondition(const std::string& vtp_file_path, const std::vector<std::string>& array_names, const StringBoolMap& flags, const faceType& face);
107
108 /// @brief Constructor for uniform values
109 /// @param uniform_values Map of array names to uniform values
110 /// @param face Face associated with the BC
111 BoundaryCondition(const StringDoubleMap& uniform_values, const StringBoolMap& flags, const faceType& face);
112
113 /// @brief Copy constructor
114 BoundaryCondition(const BoundaryCondition& other);
115
116 /// @brief Unified assignment operator (handles both copy and move)
117 BoundaryCondition& operator=(BoundaryCondition other);
118
119 /// @brief Move constructor
120 BoundaryCondition(BoundaryCondition&& other) noexcept;
121
122 /// @brief Virtual destructor
123 virtual ~BoundaryCondition() noexcept = default;
124
125 /// @brief Swap function for copy-and-swap idiom (friend function)
126 friend void swap(BoundaryCondition& lhs, BoundaryCondition& rhs) noexcept;
127
128
129 /// @brief Get value for a specific array and node
130 /// @param array_name Name of the array
131 /// @param node_id Node index on the face
132 /// @return Value for the array at the specified node
133 /// @throws std::runtime_error if array_name is not found
134 double get_value(const std::string& array_name, int node_id) const;
135
136 /// @brief Get a boolean flag by name
137 /// @param name Name of the flag
138 /// @return Value of the flag
139 /// @throws std::runtime_error if flag is not found
140 bool get_flag(const std::string& name) const;
141
142 /// @brief Get a string representation of the flags
143 /// @return String representation of the flags
144 std::string flags_to_string() const;
145
146 /// @brief Get global number of nodes
147 /// @return Global number of nodes on the face
148 int get_global_num_nodes() const noexcept {
149 return global_num_nodes_;
150 }
151
152 /// @brief Get local number of nodes
153 /// @return Local number of nodes on the face on this processor
154 int get_local_num_nodes() const noexcept {
155 return local_num_nodes_;
156 }
157
158 /// @brief Get local array index for a global node ID
159 /// @param global_node_id The global node ID defined on the face
160 /// @return Local array index for data arrays
161 /// @throws std::runtime_error if global_node_id is not found in the map
162 int get_local_index(int global_node_id) const;
163
164 /// @brief Check if data is loaded from VTP file
165 /// @return true if loaded from VTP, false if using uniform values
166 bool is_from_vtp() const noexcept {
167 return spatially_variable;
168 }
169
170 /// @brief Get the VTP file path (empty if using uniform values)
171 /// @return VTP file path
172 const std::string& get_vtp_path() const noexcept {
173 return vtp_file_path_;
174 }
175
176 /// @brief Check if this BC has been properly initialized with data
177 /// @return true if BC has data (either global or local arrays are populated)
178 bool is_initialized() const noexcept {
179 return !global_data_.empty() || !local_data_.empty();
180 }
181
182 /// @brief Distribute BC data from the master process to the slave processes
183 /// @param com_mod Reference to ComMod object for global coordinates
184 /// @param cm_mod Reference to CmMod object for MPI communication
185 /// @param cm Reference to cmType object for MPI communication
186 /// @param face Face associated with the BC
187 void distribute(const ComMod& com_mod, const CmMod& cm_mod, const cmType& cm, const faceType& face);
188
189protected:
190 /// @brief Read data from VTP file
191 /// @param vtp_file_path Path to VTP file
192 /// @param array_names Names of arrays to read
193 /// @return Map of array names to array data
194 StringArrayMap read_data_from_vtp_file(const std::string& vtp_file_path, const std::vector<std::string>& array_names);
195
196 /// @brief Find index of a point in the VTP points array
197 /// @param x X coordinate
198 /// @param y Y coordinate
199 /// @param z Z coordinate
200 /// @param vtp_points VTP points array
201 /// @return Index of the matching point in the VTP array
202 /// @throws std::runtime_error if no matching point is found
203 int find_vtp_point_index(double x, double y, double z, const Array<double>& vtp_points) const;
204
205 /// @brief Hook for derived classes to validate array values
206 /// @param array_name Name of the array being validated
207 /// @param value Value to validate
208 /// @throws std::runtime_error if validation fails
209 virtual void validate_array_value(const std::string& array_name, double value) const {}
210
211 // ---- distribute helpers ----
212 void distribute_metadata(const CmMod& cm_mod, const cmType& cm, bool is_slave);
213 void distribute_spatially_variable(const ComMod& com_mod, const CmMod& cm_mod, const cmType& cm, bool is_slave);
214 void distribute_uniform(const CmMod& cm_mod, const cmType& cm, bool is_slave);
215 void distribute_flags(const CmMod& cm_mod, const cmType& cm, bool is_slave);
216
217};

219/// @brief Base exception class for BC errors
220///
221/// These exceptions indicate fatal errors that should terminate the solver.
222/// They should not be caught and handled gracefully - the solver should exit.
223class BoundaryConditionBaseException : public std::exception {
224public:
225 /// @brief Constructor
226 /// @param msg Error message
227 explicit BoundaryConditionBaseException(const std::string& msg) : message_(msg) {}
228
229 /// @brief Get error message
230 /// @return Error message
231 const char* what() const noexcept override {
232 return message_.c_str();
233 }
234
235private:
236 std::string message_;
237};
